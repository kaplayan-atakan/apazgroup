    function mail_check(email)
    {
        var regex = /^[a-zA-Z0-9._-]+@([a-zA-Z0-9.-]+.)+([.])+[a-zA-Z0-9.-]{2,4}$/;
        if (regex.test(email)==false)
            return false;
        else
            return true;
    }



    function validate(form)
    {


        if(document.getElementById('ad_soyad').value < 3) {
            alert("Adınızı ve Soyadınızı Kontrol Ediniz");
            document.getElementById("ad_soyad").focus();
            return false;
        }

        if(document.getElementById('email').value < 3) {
            alert("Email Kontrol Ediniz");
            document.getElementById("email").focus();
            return false;
        }

        if(!mail_check(document.getElementById('email').value))
        {
            alert("Email Kontrol Ediniz");
            document.getElementById("email").focus();
            return false;
        }

        if(document.getElementById('telefon').value == "") {
            alert("Telefonunuzu Kontrol Ediniz");
            document.getElementById("telefon").focus();
            return false;
        }

        if(document.getElementById('konu').value == "") {
            alert("Konuyu Yazınız");
            document.getElementById("konu").focus();
            return false;
        }

        if(document.getElementById('mesaj').value == "") {
            alert("Mesajınızı Yazınız");
            document.getElementById("mesaj").focus();
            return false;
        }


        if(document.getElementById("kisiselVerilerinKorunmasi").checked != true) {
            alert("Kişisel Verilerin Korunmasını, Lütfen Okuyun");
            document.getElementById("kisiselVerilerinKorunmasi").focus();
            return false;
        }


        if(document.getElementById("cerezPolitikasi").checked != true) {
            alert("Çereze Politikasını, Lütfen Okuyun");
            document.getElementById("cerezPolitikasi").focus();
            return false;
        }

        return true;
    }



</script>
